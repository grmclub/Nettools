#!/bin/bash

# ==========================================
# Configuration
# ==========================================
# The command you want to run
CMD="echo 'sample output line 1' && echo 'sample output line 2'"

# The threshold count
THRESHOLD=2

# ==========================================
# Script Execution
# ==========================================

# Run the command and count the number of output lines
COUNT=$(eval "$CMD" | wc -l)

# Define color codes
GREEN='\033[0;32m'
RED='\033[0;31m'
NC='\033[0m' # No Color

echo "Command: $CMD"
echo "Counted: $COUNT | Threshold: $THRESHOLD"

# Compare and output status
if [ "$COUNT" -ge "$THRESHOLD" ]; then
    echo -e "Status: [${GREEN}PASS${NC}] - Count is $\geq$ Threshold"
else
    echo -e "Status: [${RED}FAIL${NC}] - Count is $<$ Threshold"
fi

/check for plus minus 10% of threshold

#!/bin/bash

# Configuration
CMD="ls -l"
THRESHOLD=100

# Colors
GREEN='\033[0;32m'
RED='\033[0;31m'
NC='\033[0m'

# Run command and count lines
COUNT=$($CMD | wc -l)

# Calculate 10% tolerance (integer math)
VARIANCE=$(( THRESHOLD / 10 ))
MIN=$(( THRESHOLD - VARIANCE ))
MAX=$(( THRESHOLD + VARIANCE ))

# Check if count is within the +/- 10% range
if [ "$COUNT" -ge "$MIN" ] && [ "$COUNT" -le "$MAX" ]; then
    echo -e "${GREEN}✔ PASS:${NC} Count ($COUNT) is within 10% of $THRESHOLD ($MIN to $MAX)"
else
    echo -e "${RED}✘ FAIL:${NC} Count ($COUNT) is outside 10% of $THRESHOLD ($MIN to $MAX)"
fi
