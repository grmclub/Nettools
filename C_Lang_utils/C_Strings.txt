Here are examples of some fundamental C string operations:
1. String Length (strlen)

#include <stdio.h>
#include <string.h>

int main() {
    char str[] = "Hello, World!";
    size_t length = strlen(str);
    printf("Length of string: %zu\n", length); // Output: Length of string: 13
    return 0;
}

2. String Copy (strcpy, strncpy)
C

#include <stdio.h>
#include <string.h>

int main() {
    char source[] = "Copy me!";
    char destination[20]; // Ensure destination has enough space

    // strcpy copies the entire string
    strcpy(destination, source);
    printf("Copied string (strcpy): %s\n", destination); // Output: Copied string (strcpy): Copy me!

    char partial_source[] = "Partial copy";
    char partial_destination[10]; 

    // strncpy copies a specified number of characters
    strncpy(partial_destination, partial_source, 7); // Copy first 7 characters
    partial_destination[7] = '\0'; // Null-terminate the destination string
    printf("Copied string (strncpy): %s\n", partial_destination); // Output: Copied string (strncpy): Partial
    return 0;
}

3. String Concatenation (strcat, strncat)
C

#include <stdio.h>
#include <string.h>

int main() {
    char str1[50] = "Hello, ";
    char str2[] = "World!";

    // strcat appends one string to another
    strcat(str1, str2);
    printf("Concatenated string (strcat): %s\n", str1); // Output: Concatenated string (strcat): Hello, World!

    char str3[50] = "First ";
    char str4[] = "Second part";

    // strncat appends a specified number of characters
    strncat(str3, str4, 6); // Append first 6 characters of str4
    printf("Concatenated string (strncat): %s\n", str3); // Output: Concatenated string (strncat): First Second
    return 0;
}

4. String Comparison (strcmp, strncmp)
C

#include <stdio.h>
#include <string.h>

int main() {
    char s1[] = "apple";
    char s2[] = "banana";
    char s3[] = "apple";

    // strcmp compares two strings lexicographically
    int result1 = strcmp(s1, s2); // s1 < s2, so result1 will be negative
    int result2 = strcmp(s1, s3); // s1 == s3, so result2 will be 0

    printf("Comparison of '%s' and '%s': %d\n", s1, s2, result1); // Output: Comparison of 'apple' and 'banana': (negative value)
    printf("Comparison of '%s' and '%s': %d\n", s1, s3, result2); // Output: Comparison of 'apple' and 'apple': 0

    char s4[] = "testing";
    char s5[] = "test_data";

    // strncmp compares a specified number of characters
    int result3 = strncmp(s4, s5, 4); // Compare first 4 characters ("test")
    printf("Partial comparison of '%s' and '%s' (4 chars): %d\n", s4, s5, result3); // Output: Partial comparison of 'testing' and 'test_data' (4 chars): 0
    return 0;
}

##Read String from the user

You can use the scanf() function to read a string.

The scanf() function reads the sequence of characters until it encounters whitespace (space, newline, tab, etc.).
Example 1: scanf() to read a string

#include <stdio.h>
int main()
{
    char name[20];
    printf("Enter name: ");
    scanf("%s", name);
    printf("Your name is %s.", name);
    return 0;
}

##Strings and Pointers

Similar like arrays, string names are "decayed" to pointers. Hence, you can use pointers to manipulate elements of the string. We recommended you to check C Arrays and Pointers before you check this example.
Example 4: Strings and Pointers

#include <stdio.h>

int main(void) {
  char name[] = "Harry Potter";

  printf("%c", *name);     // Output: H
  printf("%c", *(name+1));   // Output: a
  printf("%c", *(name+7));   // Output: o

  char *namePtr;

  namePtr = name;
  printf("%c", *namePtr);     // Output: H
  printf("%c", *(namePtr+1));   // Output: a
  printf("%c", *(namePtr+7));   // Output: o
}

##Parse a Key-Value string

#include <stdio.h>
#include <string.h>
#include <stdlib.h> // For malloc and free

void parse_key_value(const char *input_string, char **key, char **value) {
    char *equals_sign = strchr(input_string, '=');

    if (equals_sign == NULL) {
        *key = NULL;
        *value = NULL;
        return;
    }

    // Allocate memory for the key
    size_t key_len = equals_sign - input_string;
    *key = (char *)malloc(key_len + 1);
    if (*key == NULL) {
        perror("Failed to allocate memory for key");
        exit(EXIT_FAILURE);
    }
    strncpy(*key, input_string, key_len);
    (*key)[key_len] = '\0'; // Null-terminate the key

    // Allocate memory for the value
    size_t value_len = strlen(equals_sign + 1);
    *value = (char *)malloc(value_len + 1);
    if (*value == NULL) {
        perror("Failed to allocate memory for value");
        exit(EXIT_FAILURE);
    }
    strcpy(*value, equals_sign + 1); // Copy the value
}

int main() {
    const char *test_string = "name=John Doe";
    char *parsed_key = NULL;
    char *parsed_value = NULL;

    parse_key_value(test_string, &parsed_key, &parsed_value);

    if (parsed_key != NULL && parsed_value != NULL) {
        printf("Key: %s\n", parsed_key);
        printf("Value: %s\n", parsed_value);
    } else {
        printf("Failed to parse key-value pair.\n");
    }

    // Free allocated memory
    free(parsed_key);
    free(parsed_value);

    return 0;
}
